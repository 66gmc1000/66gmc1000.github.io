Hunt for broken Customer Request Types in Jira Service Desk Jira Service Management

June 16, 2021 2-minute read
blog
scripts • bash • oneliners • jira

Looks for conditions where a customer request type is not set in a project or set to a blank value and outputs a JQL search for identifying these issues in the Advanced Issue Search

$Text = '<youratlassianusernamehere>:<youratlassianapitokenhere>'
$Bytes = [System.Text.Encoding]::UTF8.GetBytes($Text)
$EncodedText = [Convert]::ToBase64String($Bytes)
$EncodedText

$projectKey = 'DIGI'

$jqlQuery = "project = $projectKey"
$jqlEncoded = [System.Web.HttpUtility]::UrlEncode("$jqlQuery")
$Text = '<youratlassianusernamehere>:<youratlassianapitokenhere>'
$Bytes = [System.Text.Encoding]::UTF8.GetBytes($Text)
$EncodedText = [Convert]::ToBase64String($Bytes)
$EncodedText
$Uri = "https://wecu.atlassian.net/rest/api/3/search?jql=$jqlEncoded"
$Method = "GET"
$headers = @{
    'Authorization' = "Basic $EncodedText"
    'Content-Type' = 'application/json'
}
$Response = Invoke-RestMethod -Uri $Uri -Method $Method -Headers $headers

$totalResults = $Response.total
$maxResults = $Response.maxResults
$startAt = '1'

$QueryResults = @()

do {
    $Results = Invoke-RestMethod -Uri $Uri -Method $Method -Headers $headers
    if ($Results.issues) {
        $QueryResults += $Results.issues
        $startHere = $Results | select -ExpandProperty startAt
        $startAt = [int]$startHere + [int]$maxResults
    }
    else {
        $QueryResults += $Results
    }
    $uri = "https://wecu.atlassian.net/rest/api/3/search?jql=$jqlEncoded&fields=id,key&startAt=$startAt"
} until (!($startAt -lt $totalResults))

$Keys = $QueryResults | select-object -ExpandProperty Key

$huntResults = @()

foreach ($Key in $Keys) {
    $Uri = "https://wecu.atlassian.net/rest/api/2/issue/$Key&fields=customfield_10001,assignee,reporter,summary,key,id,status,issuetype"
    $Method = "GET"
    $headers = @{
        'Authorization' = "Basic $EncodedText"
        'Content-Type' = 'application/json'
    }

    $Response = Invoke-RestMethod -Uri $Uri -Method $Method -Headers $headers
    $CustomerRequestTypeName = $Response.fields.customfield_10001.requestType.name
    $CustomerRequestTypeId = $Response.fields.customfield_10001.requestType.id
    $Assignee = $Response.fields.assignee.displayname
    $Reporter = $Response.fields.reporter.displayName
    $Summary = $Response.fields.summary
    $IssueKey = $Response.key
    $Status = $Response.fields.status.name
    $IssueType = $Response.fields.issueType.name

        $huntResults+=new-object psobject -Property @{
            "Key" = "$IssueKey";
            "IssueType" = "$IssueType";
            "Summary" = "$Summary";
            "Status" = "$Status";
            "Reporter" = "$Reporter";
            "Assignee" = "$Assignee";
            "CustomerRequestTypeName" = "$CustomerRequestTypeName";
            "CustomerRequestTypeId" = "$CustomerRequestTypeId";
            }
}

($huntResults).count

$huntResults

$Other = $huntResults | where IssueType -eq "Support Ticket - $projectKey" | where CustomerRequestTypeName -eq ""

$Rows = @()
$Rows += $Other.Key -join ","

$Rows
